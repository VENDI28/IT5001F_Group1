# ANSWER: Handle missing values
num_cols = df_orders_full.select_dtypes(include="number").columns
df_orders_full[num_cols] = df_orders_full[num_cols].apply(lambda x: x.fillna(x.median())) # Fill with the median value for numeric columns

df_orders_full = df_orders_full.dropna(subset=["order_id", "payment_type"])
